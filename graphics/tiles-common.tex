\tikzset{
	logic tile/.style = { violet!30 },
	ram tile/.style = { yellow!30 },
	io tile/.style = { cyan!30 },
	io pin/.style = { },
	unused io pin/.style = { lightgray }
}

\newcommand{\logictile}[2]{
  \filldraw[logic tile] (#1 - .45, #2 - .45) rectangle (#1 + .45, #2 + .45);
  \draw (#1, #2 + .15) node {\sf logic};
  \draw (#1, #2 - .2)  node {\sf #1{} #2};
}
\newcommand{\iotile}[2]{
  \filldraw[io tile] (#1 - .45, #2 - .45) rectangle (#1 + .45, #2 + .45);
  \draw (#1, #2 + .18) node {\sf io};
  \draw (#1, #2 - .2)  node {\sf #1{} #2};
}
\newcommand{\ramtile}[3]{
  \filldraw[ram tile] (#1 - .45, #2 - .45) rectangle (#1 + .45, #2 + 1.45);
  \draw (#1, #2 +  .18) node {\sf ramb};
  \draw (#1, #2 -   .2) node {\sf #1{} #2};
  \draw (#1, #2 + 1.16) node {\sf ramt};
  \draw (#1, #2 +   .8) node {\sf #1{} #3};
}

\newcommand{\leftpin}[5]{
  \draw [#5 io pin, line width=.1mm]
	(#1 - .45, #2 - .25 + #3 * .5) --
	(#1 -  .9, #2 - .25 + #3 * .5);
  \draw [#5 io pin, line width=2.5mm]
	(#1 -  .9, #2 - .25 + #3 * .5) --
	(#1 - 1.5, #2 - .25 + #3 * .5);
  \draw [#5 io pin]
	(#1 - 1.6, #2 - .25 + #3 * .5) node [anchor=east] {\sf #4};
}
\newcommand{\rightpin}[5]{
  \draw [#5 io pin, line width=.1mm]
	(#1 + .45, #2 - .25 + #3 * .5) --
	(#1 +  .9, #2 - .25 + #3 * .5);
  \draw [#5 io pin, line width=2.5mm]
	(#1 +  .9, #2 - .25 + #3 * .5) --
	(#1 + 1.5, #2 - .25 + #3 * .5);
  \draw [#5 io pin]
	(#1 + 1.6, #2 - .25 + #3 * .5) node [anchor=west] {\sf #4};
}
\newcommand{\toppin}[5]{
  \draw [#5 io pin, line width=.1mm]
	(#1 - .25 + #3 * .5, #2 + .45) --
	(#1 - .25 + #3 * .5, #2 +  .9);
  \draw [#5 io pin, line width=2.5mm]
	(#1 - .25 + #3 * .5, #2 +  .9) --
	(#1 - .25 + #3 * .5, #2 + 1.5);
  \draw [#5 io pin]
	(#1 - .25 + #3 * .5, #2 + 1.6) node [anchor=east, rotate=-90] {\sf #4};
}
\newcommand{\bottompin}[5]{
  \draw [#5 io pin, line width=.1mm]
	(#1 - .25 + #3 * .5, #2 - .45) --
	(#1 - .25 + #3 * .5, #2 - .9);
  \draw [#5 io pin, line width=2.5mm]
	(#1 - .25 + #3 * .5, #2 - .9) --
	(#1 - .25 + #3 * .5, #2 - 1.5);
  \draw [#5 io pin]
	(#1 - .25 + #3 * .5, #2 - 1.6) node [anchor=west, rotate=-90] {\sf #4};
}
